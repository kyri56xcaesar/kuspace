- convert to mp4
  -c:v libx264 -preset fast -crf 23

- extract audio
  -q:a 0 -map a -vn

- resize to 720p
  -vf scale=1280:720

- trim video 
 -ss 00:00:05 -to 00:00:10 -c copy

- covert to gif
 -vf "fps=10,scale=320:-1" -loop 0 output.gif
